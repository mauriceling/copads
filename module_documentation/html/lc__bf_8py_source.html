1 '''
2 Loose Circular Brainfuck (LCBF) Interpreter
3 Date created: 15th August 2012
4 Licence: Python Software Foundation License version 2
5 
6 LCBF uses all 8 operations in standard Brainfuck with the
7 following differences:
8  1. The tape or array is circular (a ring list) instead of
9  linear. When the pointer is at the "end" of the tape, an
10  increment (">") will move the tape to the start. Similarly,
11  when the pointer is decremented at the "beginning" of the
12  tape, the pointer goes to the end.
13  2. Operations after a start loop operator ("[") will only be
14  executed provided the loop(s) are properly closed. If the
15  loops are open, the program will terminate.
16  3. However, it is possible to have an end loop operator ("]")
17  without a preceding start loop operator ("["). In this case,
18  the end loop operator ("]") will be ignored and execution continues.
19  4. Unclosed or unopened loops may result in non-deterministic
20  behaviour.
21  5. All inputs are pre-defined at the start of the program.
22 
23 @see: http://esolangs.org/wiki/Loose_Circular_Brainfuck_(LCBF)
24 '''

72 def cbf_start_loop(array, apointer, inputdata, output, source, spointer):
73  '''
74  Start loop. Operations after a start loop operator ("[") will only
75  be executed provided the loop(s) are properly closed. If the loops
76  are open, the program will terminate. Note that unclosed or unopened
77  loops may result in non-deterministic behaviour.
78  '''
79  if array[apointer] > 0:
80  return (array, apointer, inputdata, output, source, spointer)
81  else:
82  count = 1
83  try:
84  while count > 0:
85  spointer = spointer + 1
86  if source[spointer] == ']':
87  count = count - 1
88  if source[spointer] == '[':
89  count = count + 1
90  except IndexError:
91  spointer = len(source) - 1
92  return (array, apointer, inputdata, output, source, spointer)

94 def cbf_end_loop(array, apointer, inputdata, output, source, spointer):
95  '''
96  End loop. However, it is possible to have an end loop operator
97  ("]") without a preceding start loop operator ("["). In this case,
98  the end loop operator ("]") will be ignored and execution continues.
99  Note that unclosed or unopened loops may result in non-deterministic
100  behaviour.
101  '''
102  temp = spointer
103  if array[apointer] < 1:
104  return (array, apointer, inputdata, output, source, spointer + 1)
105  else:
106  count = 1
107  try:
108  while count > 0:
109  spointer = spointer - 1
110  if source[spointer] == ']':
111  count = count + 1
112  if source[spointer] == '[':
113  count = count - 1
114  except IndexError:
115  spointer = temp
116  return (array, apointer, inputdata, output, source, spointer)
