42  def rmutate(self, type='point', rate=0.01, start=0, end=-1):
43  """
44  Random Mutation operator - to simulate random point, insertion,
45  deletion, inversion, gene translocation and gene duplication events.
46 
47  The start and end parameters are useful for simulating mutational
48  hotspots in the genome.
49 
50  @param type: type of mutation. Accepts 'point' (point mutation),
51  'insert' (insert a base), 'delete' (delete a base), 'invert'
52  (invert a stretch of the chromosome), 'duplicate' (duplicate a
53  stretch of the chromosome), 'translocate' (translocate a stretch of
54  chromosome to another random position). Default = point.
55  @param rate: probability of mutation per base above background
56  mutation rate. Default = 0.01 (1%). No mutation event will ever
57  happen if (rate + background_mutation) is less than zero.
58  @param start: starting base on the sequence for mutation.
59  Default = 0, start of the genome.
60  @param end: last base on the sequence for mutation. Default = -1, end of
61  the genome.
62 
63  @since: version 0.4
64  """
65  if end > len(self.sequence) - 1: end = len(self.sequence) - 1
66  if end == -1: end = len(self.sequence) - 1
67  if start == end: start = 0
68  length = int(end - start)
69  mutation = int((self.background_mutation + rate) * length)
70  while mutation > 0:
71  position = int(start) + random.randrange(length - 1)
72  new_base = self.base[random.randrange(len(self.base))]
73  if type == 'point':
74  self.sequence[position] = new_base
75  if type == 'delete':
76  self.sequence.pop(position)
77  if type == 'insert':
78  self.sequence.insert(position, new_base)
79  if type == 'duplicate':
80  end_pos = random.randrange(position + 1, end)
81  fragment = self.sequence[position:end_pos]
82  for i in range(len(fragment)):
83  self.sequence.insert(end_pos + i, fragment[i])
84  if type == 'invert':
85  end_pos = random.randrange(position + 1, end)
86  fragment = [self.sequence.pop(position)
87  for i in range(end_pos - position)]
88  fragment.reverse()
89  for base in fragment:
90  self.sequence.insert(position, base)
91  if type == 'translocate':
92  end_pos = random.randrange(position + 1, end)
93  fragment = [self.sequence.pop(position)
94  for i in range(end_pos - position)]
95  insertion_point = random.randint(0, len(self.sequence))
96  for i in range(len(fragment)):
97  self.sequence.insert(insertion_point + i, fragment[i])
98  mutation = mutation - 1

100  def kmutate(self, type='point', start=0, end=0, sequence=None, tpos=0):
101  """
102  Known Mutation operator - to simulate a known point, insertion,
103  deletion, inversion, gene translocation or gene duplication event.
104 
105  The required parameters will be determined by the type of mutation:
106  - point requires
107  - start (the position of the base to mutate)
108  - sequence (the new base)
109  - delete requires
110  - start (the position of the base to delete)
111  - insert requires
112  - start (the position of the base to begin insertion)
113  - sequence (list of sequence or a base to insert)
114  - invert requires
115  - start (the position of the base to start inversion)
116  - end (the position of the base to end inversion)
117  - duplicate requires
118  - start (the position of the starting base to duplicate)
119  - end (the position of the last base to duplicate)
120  - translocate requires
121  - start (the position of the base to start translocation)
122  - end (the position of the base to end translocation)
123  - tpos (the position of the base insert the translocated
124  sequence)
125 
126  @param type: type of mutation. Accepts 'point' (point mutation),
127  'insert' (insert a base), 'delete' (delete a base), 'invert'
128  (invert a stretch of the chromosome), 'duplicate' (duplicate a
129  stretch of the chromosome), 'translocate' (translocate a stretch of
130  chromosome to another random position). Default = point.
131  @param start: starting base on the chromosome for mutation.
132  Default = 0, start of the genome.
133  @param end: last base on the chromosome for mutation. Default = 0.
134  @param sequence: list of bases to change to (point mutation) or to
135  insert (insertion mutation)
136  @param tpos: position of the chromosome to insert the translocated
137  sequence.
138 
139  @since: version 0.4
140  """
141  if type == 'point':
142  self.sequence[start] = sequence
143  if type == 'delete':
144  self.sequence.pop(start)
145  if type == 'insert':
146  for base in list(sequence):
147  self.sequence.insert(start, base)
148  if type == 'duplicate':
149  fragment = self.sequence[start:end + 1]
150  for i in range(len(fragment)):
151  self.sequence.insert(end + i, fragment[i])
152  if type == 'invert':
153  for base in self.sequence[start:end]:
154  self.sequence.insert(start, base)
155  if type == 'translocate':
156  fragment = [self.sequence.pop(start)
157  for i in range(end - start)]
158  for i in range(len(fragment)):
159  self.sequence.insert(tpos + i, fragment[i])

253  substituted as fitness function is highly dependent on utility.} The
254  only requirement is that a fitness score must be returned.

259  @return: fitness score or fitness list

269  def mutation_scheme(self, type=None, rate=None):
270  """
271  Function to trigger mutation events in each chromosome. B{This function
272  may be over-ridden by the inherited class or substituted to cater for
273  specific mutation schemes but not an absolute requirement to do so.}
274 
275  Both type and rate must be defined at the same time, otherwise the
276  initiated mutation_type and additional_mutation_rate will be used.
277 
278  @param type: type of mutation. Accepts 'point' (point mutation),
279  'insert' (insert a base), 'delete' (delete a base), 'invert'
280  (invert a stretch of the chromosome), 'duplicate' (duplicate a
281  stretch of the chromosome), 'translocate' (translocate a stretch of
282  chromosome to another random position). Default = None.
283  @param rate: probability of mutation per base above background mutation
284  rate. Default = None. No mutation event will ever happen if
285  (rate + background_mutation) is less than zero.
286 
287  @since: version 0.4
288  """
289  for chromosome in self.genome:

296  def setStatus(self, variable, value):
297  """
298  Sets new status or change status of the organism. However, the following
299  status change will result in death of the organism
300  1. 'alive' to False
301  2. 'vitality' to or below zero
302  3. 'age' to or above lifespan
303 
304  @param variable: name of status to change
305  @param value: new value of the status
306 
307  @since: version 0.4
308  """
309  if variable == 'alive' and value == True:
310  self.status['alive'] = value
311  if variable == 'alive' and value == False:
312  self.status['alive'] = value
313  self.status['death'] = 'death03'
314  if variable == 'vitality':
315  if float(value) > 100.1:
316  self.status['vitality'] = 100.0
317  if float(value) > 0.0 and float(value) < 100.1:
318  self.status['vitality'] = float(value)
319  if float(value) == 0 or float(value) < 0:
320  self.status['vitality'] = 0
321  self.status['alive'] = False
322  self.status['death'] = 'death01'
323  elif variable == 'age':
324  if float(value) < self.status['lifespan']:
325  self.status['age'] = float(value)
326  else:
327  self.status['age'] = self.status['lifespan']
328  self.status['alive'] = False
329  self.status['death'] = 'death02'
330  else:
331  self.status[variable] = value

399  def prepopulation_control(self):
400  """
401  Function to trigger population control events before mating event in
402  each generation (For example, to simulate pre-puberty death). B{This
403  function may be over-ridden by the inherited class or substituted to
404  cater for specific events.} Although this is not an absolute
405  requirement, it is extremely encouraged to prevent exhaustion of memory
406  space. Without population control, it will seems like a reproducing
407  immortal population.
408 
409  Here, the sample implementation eliminates the bottom half of the
410  population based on fitness score unless the number of organisms is
411  less than 20. This is to prevent extinction. However, if the number of
412  organisms is more than 2000 (more than 100x initial population size,
413  a random selection of 2000 will be used for the next generation.
414 
415  @since: version 0.4
416  """
417  size = len(self.agents)
418  sfitness = [self.agents[x].fitness() for x in range(size)]
419  threshold = sum(sfitness) / len(sfitness)
420  temp = [self.agents[x]
421  for x in range(size)
422  if sfitness[x] > threshold]
423  if len(temp) > 2001:
424  size = len(temp)
425  self.agents = [temp[random.randint(0, size - 1)]
426  for x in xrange(2000)]
427  if len(temp) > 21:
428  self.agents = temp

430  def mating(self):
431  """
432  Function to trigger mating events in each generation. B{This function
433  may be over-ridden by the inherited class or substituted to cater for
434  specific mating schemes but not an absolute requirement to do so.}
435  Mating schemes should include
436  - selection of mating partners using Organism.fitness() function
437  and/or other status
438  - processes and actions of mating
439 
440  Here, the sample implementation randomly selects any 2 organisms from
441  the culled list and generate a new genome for the progeny organism by
442  single crossover.
443 
444  @since: version 0.4
445  """
446  size = len(self.agents)
447  temp = []
448  for x in xrange(size):
449  organism1 = self.agents[random.randint(0, size - 1)]
450  organism2 = self.agents[random.randint(0, size - 1)]
451  crossover_pt = random.randint(0, len(organism1.genome[0].sequence))
452  (g1, g2) = crossover(organism1.genome[0], organism2.genome[0],
453  crossover_pt)
454  temp = temp + [Organism([g1])]
455  self.add_organism(temp)

502  def generation_step(self):
503  """
504  Function to simulate events for one generation. These includes
505  - mating (according to the mating scheme or function)
506  - mutating each organism in the population
507  - population size control

510  - reporting the population status

512  @return: information returned from report function.
513

517  self.prepopulation_control()

519  self.postpopulation_control()
520  for organism in self.agents:

579 def crossover(chromosome1, chromosome2, position):
580  """
581  Cross-over operator - swaps the data on the 2 given chromosomes after
582  a given position.
583 
584  @param chromosome1: Chromosome object
585  @param chromosome2: Chromosome object
586  @param position: base position of the swap over
587  @type position: integer
588  @return: (resulting chromosome1, resulting chromosome2)
589 
590  @since: version 0.4
591  """
592  seq1 = chromosome1.sequence
593  seq2 = chromosome2.sequence
594  position = int(position)
595  if len(seq1) > position and len(seq2) > position:
596  new1 = Chromosome(seq1[:position] + seq2[position:],
597  chromosome1.base)
598  new2 = Chromosome(seq2[:position] + seq1[position:],
599  chromosome2.base)
600  return (new1, new2)
601  elif len(seq1) > position:
602  new1 = Chromosome(seq1[:position], chromosome1.base)
603  new2 = Chromosome(chromosome2.sequence + seq1[position:],
604  chromosome2.base)
605  return (new1, new2)
606  elif len(seq2) > position:
607  new1 = Chromosome(chromosome1.sequence + seq2[position:],
608  chromosome1.base)
609  new2= Chromosome(seq2[:position], chromosome2.base)
610  return (new1, new2)
611  else:
612  return (chromosome1, chromosome2)
