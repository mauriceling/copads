41  def _check_brain(self):
42  '''
43  Private method - Checks for the presence of tables in the brain file,
44  as a proxy for checking whether the supplied brain file (in
45  constructor) is a new file or an existing file. If it is new brain,
46  then the tables need to be created.
47  '''
48  results = [True, True, True, True]
49  try:
50  self.cur.execute('select max(ID) from neuron')
51  except:
52  results[0] = False
53  try:
54  self.cur.execute('select max(ID) from neuron_internal_states')
55  except:
56  results[1] = False
57  try:
58  self.cur.execute('select max(ID) from neuron_input_states')
59  except:
60  results[2] = False
61  try:
62  self.cur.execute('select max(source) from connectome')
63  except:
64  results[3] = False
65  return results

67  def _format_brain(self):
68  '''
69  Private method - Setting up the tables needed for a new brain file.
70  '''
71  try:
72  self.cur.execute('drop table logfile')
73  except s.OperationalError:
74  pass
75  self.cur.execute('''
76  create table if not exists logfile (
77  ID integer primary key autoincrement,
78  event text not null)''')
79  try:
80  self.cur.execute('drop table neuron')
81  except s.OperationalError:
82  pass
83  self.cur.execute('''
84  create table if not exists neuron (
85  ID integer primary key autoincrement,
86  name text not null,
87  status text not null default 'alive')''')
88  try:
89  self.cur.execute('drop table neuron_internal_states')
90  except s.OperationalError:
91  pass
92  self.cur.execute('''
93  create table if not exists neuron_internal_states (
94  ID integer not null,
95  key text not null,
96  value text not null,
97  primary key (ID, key))''')
98  try:
99  self.cur.execute('drop table neuron_input_states')
100  except s.OperationalError:
101  pass
102  self.cur.execute('''
103  create table if not exists neuron_input_states (
104  ID integer not null,
105  sourceID text not null,
106  value text not null,
107  primary key (ID, sourceID))''')
108  try:
109  self.cur.execute('drop table connectome')
110  except s.OperationalError:
111  pass
112  self.cur.execute('''
113  create table if not exists connectome (
114  source integer not null,
115  destination integer not null,
116  key text not null default 'weight',
117  value text not null,
118  primary key (source, destination, key))''')
119  self.conn.commit()

134  def add_neuron(self, **kwargs):
135  '''
136  Method to add a new neuron into the brain.
137 
138  Logging messages::
139  - add neuron: ADDNEURON:<ID of new neuron>:<name of new neuron>
140  - set initial neuron internal data: SETININED:<ID of new neuron>:
141  <initial neuronal data in dictionary string format>
142 
143  @param kwargs: initial neuronal data (as dictionary). The following
144  mandatory fields will be added for each neuron::
145  - name: user-defined name for the neuron. Default = no_name.
146  - summation: type of summation functions. Allowable values are
147  'summation' (averaged input values). Default = summation.
148  - transfer: type of transfer / activation functions, which will
149  generate the output value of the neuron. Allowable values are
150  'linear' (neuron output = output from summation function),
151  'sigmoid' (sigmoid curve output). Default = linear.
152  - threshold: theshold value of the neuron. Default = 0.5.
153  - pstate: initial output value of the neuron. Default = generated
154  random value between 0-1.
155  @type kwargs: dictionary
156  @return: ID of the new neuron
157  '''
158  if 'name' not in kwargs:
159  name = 'no_name'
160  else:
161  name = str(kwargs['name'])
162  del kwargs['name']
163  if 'summation' not in kwargs:
164  kwargs['summation'] = 'summation'
165  if 'transfer' not in kwargs:
166  kwargs['transfer'] = 'linear'
167  if 'threshold' not in kwargs:
168  kwargs['threshold'] = '0.5'
169  if 'pstate' not in kwargs:
170  kwargs['pstate'] = str(random())
171  self.cur.execute('insert into neuron (name) values (?)', (name,))
172  ID = str(self.cur.lastrowid)
173  for key in kwargs:
174  self.cur.execute('''
175  insert into neuron_internal_states (ID, key, value)
176  values (?,?,?)''', (ID, str(key), str(kwargs[key])))
177  self.log('ADDNEURON:' + ID + ':' + name)
178  self.log('SETININED:' + ID + ':' + str(kwargs))
179  self.conn.commit()
180  return ID

182  def delete_neuron(self, ID):
183  '''
184  Method to delete a neuron from the brain and all associated synapses.
185 
186  Logging messages::
187  - delete source synapses: DELNES:<ID of neuron to delete>
188  - delete destination synapses: DELNED:<ID of neuron to delete>
189  - delete internal states: DELNIS:<ID of neuron to delete>
190  - delete input states: DELNIN:<ID of neuron to delete>
191  - delete neuron: DELNEU:<ID of neuron to delete>
192 
193  @param ID: ID of neuron to delete
194  @type ID: string
195  '''
196  self.cur.execute('''delete from connectome where source=:ID''',
197  {'ID': str(ID)})
198  self.log('DELNES:' + ID)
199  self.cur.execute('''delete from connectome where destination=:ID''',
200  {'ID': str(ID)})
201  self.log('DELNED:' + ID)
202  self.cur.execute('''delete from neuron_internal_states where ID=:ID''',
203  {'ID': str(ID)})
204  self.log('DELNIS:' + ID)
205  self.cur.execute('''delete from neuron_input_states where ID=:ID''',
206  {'ID': str(ID)})
207  self.log('DELNIN:' + ID)
208  self.cur.execute('''delete from neuron where ID=:ID''',
209  {'ID': str(ID)})
210  self.log('DELNEU:' + ID)
211  self.conn.commit()

228  def add_synapse(self, sourceID, destinationID):
229  '''
230  Method to add a new synapse (connection between 2 neurons) into the
231  brain. When either of the neurons are not present in the brain, 1 or 2
232  neurons (depending whether 1 or both neurons are not found) will be
233  created; however, the ID(s) of the neuron(s) will not be the same as
234  provided. A random weight between 0 to 1 (as synaptic state) will be
235  added.
236 
237  Logging messages::
238  - add synapse: ADDSYN:<ID of source / originating neuron>:<ID of
239  destination / sink neuron>:<initial synaptic weight in dictionary
240  string format>
241 
242  @param sourceID: ID of source / originating neuron.
243  @type sourceID: string
244  @param destinationID: ID of destination / sink neuron.
245  @type destinationID: string
246  '''
247  if self._check_neuron_presence(sourceID) == False:
248  sourceID = self.add_neuron()
249  if self._check_neuron_presence(destinationID) == False:
250  destinationID = self.add_neuron()
251  kwargs = {'weight': random()}
252  for key in kwargs:
253  self.cur.execute('''
254  insert into connectome (source, destination, key, value)
255  values (?,?,?,?)''',
256  (sourceID, destinationID, str(key), str(kwargs[key])))
257  self.log('ADDSYN:' + sourceID + ':' + destinationID + ':' + str(kwargs))
258  self.conn.commit()

718  def _neuron_summation(self, ID):
719  '''
720  Private method - Calculate a single input (to be used as input for
721  transfer / activation function) from a set of input vectors. If there
722  is no summation function for the neuron or if the calculation returns
723  an error (exception), 0.0 will be returned.
724 
725  @param ID: ID of neuron to calculate.
726  @type ID: string
727  @return: calculated input.
728  '''

731  return 0.0

733  neuron_input = self.base.get_input_values(ID)

735  if sum_func == 'summation':

740  def _neuron_transfer(self, ID, neuron_input):
741  '''
742  Private method - Calculate the neuron output, using the transfer /
743  activation function, from the calculated input of the summation
744  function. If there is no transfer / activation function for the neuron
745  or if the calculation returns an error (exception), 0.0 will be
746  returned.
747 
748  @param ID: ID of neuron to calculate.
749  @type ID: string
750  @param neuron_input: output from _neuron_summation(), to be used as
751  input for transfer / activation function.
752  @type neuron_input: float
753  @return: output of transfer / activation function.
754  '''
755  transfer = self.base.get_transfer(ID)
756  if transfer == None:
757  return 0.0
758  transfer = str(transfer[0])
759  try:
760  if transfer == 'linear':
761  return neuron_input
762  elif transfer == 'sigmoid':
763  return 1.0 / (1.0 + math.exp(-1.0 * neuron_input))
764  except:
765  return 0.0

789  def execute_neuron(self, ID, state='pstate', stype='weight'):
790  '''
791  Method to execute a neuron.
792 
793  The following tasks will be executed for a neuron::
794  1. calculate a value from the set of input value vectors

796  activation function

987  def delete_synapse_by_state(self, threshold=0.1, mode='lowest',
988  stype='weight'):
989  '''
990  Method to delete synapse(s) by thresholding on synaptic states. There
991  are a number of ways (criterion) to select synapses for deletion. The
992  criterion is determined by the 'mode' parameter.
993 
994  The following criteria (modes) are allowed::
995  - lowest: select the lowest percentage of synapses, by state value,
996  for deletion. Threshold will determine the ratio. For example, if
997  threshold = 0.1, means the lowest decile of the synapses, by state
998  value, will be deleted.
999  - below: select the synapses with state value below the threshold
1000  value for deletion. For example, if threshold = 0.1, means that
1001  synapses with state value lower than 0.1, will be deleted.
1002 
1003  @param threshold: threshold for use in deletion (as ratio). Default =
1004  lowest.
1005  @type threshold: float
1006  @param mode: criteria for deletion. Default = lowest.
1007  @type mode: string
1008  @param stype: synaptic state for use in deletion criterion. Default =
1009  weight.
1010  @type stype: float
1011  '''
1012  states = [float(x) for x in self.get_all_synaptic_states(stype)]
1013  states.sort()
1014  if mode == 'lowest':
1015  states = states[:int(len(states)*float(threshold))]
1016  elif mode == 'below':
1017  states = [x for x in states if x < float(threshold)]
1018  for value in states:
1019  self.base.cur.execute('''
1020  select source, destination from connectome where
1021  key=:key and value=:value''',
1022  {'key': str(stype),
1023  'value': str(value)})
1024  result = self.base.cur.fetchone()
1025  sourceID = str(result[0])
1026  destinationID = str(result[1])
1027  self.delete_synapse(sourceID, destinationID)
