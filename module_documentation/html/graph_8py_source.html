52  def makeGraphFromAdjacency(self, adj):
53  """
54  Constructs a graph from an adjacency (adj) matrix, which is given
55  as a list of list (rows).
56  The first row of the matrix contains a list of the vertices; hence,
57  there will be n+1 rows and n-columns in the given matrix.
58 
59  @param adj: adjacency matrix, with vertices as the first row -
60  row count is 1 more than column count
61  @type adj: list of list
62  """
63  vertices = adj.pop(0)

66  ends = {}

68  for col in range(len(adj[row])):
69  if adj[row][col] > 0: ends[vertices[col]] = adj[row][col]
70  self.graph[vertices[row]] = ends

86  def makeGraphFromEdges1(self, edges):

92  @param edges: edges

95  @status: Tested method

97  """

115  Constructs an un-directional graph from edges (a list of tuple).
116  Each tuple contains 2 vertices.

123  list of tuples')

130  row = vertices.index(e[0])
131  col = vertices.index(e[1])

155  def Dijkstra(self, start, end=None):
156  """
157  Find shortest paths from the start vertex to all vertices nearer than
158  or equal to the end.
159 
160  Dijkstra's algorithm is only guaranteed to work correctly when all
161  edge lengths are positive. This code does not verify this property
162  for all edges (only the edges seen before the end vertex is reached),
163  but will correctly compute shortest paths even for some graphs with
164  negative edges, and will raise an exception if it discovers that a
165  negative edge has caused it to make a mistake.
166 
167  @param start: vertex of starting point
168  @param end: vertex of ending point
169 
170  @status: Tested method (by proxy from testing shortestPath method)
171  @since: version 0.1
172  """
173  D = {} # dictionary of final distances
174  P = {} # dictionary of predecessors
175  Q = PriorityDictionary() # est.dist. of non-final vert.
176  Q[start] = 0
177  for v in Q:
178  D[v] = Q[v]
179  if v == end: break
180  try: self.graph[v]
181  except KeyError: continue
182  for w in self.graph[v]:
183  if self.graph[v][w] > 0:
184  vwLength = D[v] + self.graph[v][w]
185  if w in D:
186  if vwLength < D[w]:
187  raise ValueError("Dijkstra: found better path to \
188  already-final vertex")
189  elif w not in Q or vwLength < Q[w]:
190  Q[w] = vwLength
191  P[w] = v
192  return (D, P)
