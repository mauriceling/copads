39  national characters are used on canvas.
40

78  state = options['state']

81  elif options['state'] == '':
82  options['state'] = 'normal'
83  else:

91  if state == DISABLED and options.get(state + name):
92  return options.get(state + name)
93

95  return options.get(name)
96  else:
97  return default

101  options['outline'] = ''
102  options['activeoutline'] = ''
103  options['disabledoutline'] = ''

105  options['fill'] = ''
106  options['activefill'] = ''
107  options['disabledfill'] = ''

111

114  else:

121  elif state == DISABLED:

123  width = options['disabledwidth']
124

126  style['stroke-width'] = width
127 
128

131  if state == DISABLED and canvas.itemcget(item, 'disableddash'):
132  dash = canvas.itemcget(item, 'disableddash')
133  elif state == ACTIVE and canvas.itemcget(item, 'activedash'):
134  dash = canvas.itemcget(item, 'activedash')
135

143

145  style['stroke-dashoffset'] = options['dashoffset']
146 
147 
148  if itemtype == 'line':
149  # in this case, outline is set with fill property
150  style["fill"], style["stroke"] = "none", style["fill"]

237  style['font-family'] = actual['family']

350 def smoothline(document, coords):
351  "smoothed polyline"
352  element = document.createElement('path')
353  path = []
354 
355  points = [(coords[i], coords[i+1]) for i in range(0, len(coords), 2)]
356  def pt(points):
357  x0, y0 = points[0]
358  x1, y1 = points[1]
359  p0 = (2*x0-x1, 2*y0-y1)
360 
361  x0, y0 = points[-1]
362  x1, y1 = points[-2]
363  pn = (2*x0-x1, 2*y0-y1)
364 
365  p = [p0] + points[1:-1] + [pn]
366 
367  for i in range(1, len(points)-1):
368  a = p[i-1]
369  b = p[i]
370  c = p[i+1]
371 
372  yield lerp(a, b, 0.5), b, lerp(b, c, 0.5)
373 
374 
375  for i, (A, B, C) in enumerate(pt(points)):
376  if i == 0:
377  path.append("M%s,%s Q%s,%s %s,%s" % (A[0], A[1], B[0], B[1], C[0], C[1]))
378  else:
379  path.append("T%s,%s" % (C[0], C[1]))
380 
381  element.setAttribute('d', ' '.join(path))
382  return element

469 def arc(document, bounding_rect, start, extent, style):
470  "arc, pieslice (filled), arc with chord (filled)"
471  (x1, y1, x2, y2) = bounding_rect
472  import math
473 
474  cx = (x1 + x2)/2.0
475  cy = (y1 + y2)/2.0
476 
477  rx = (x2 - x1)/2.0
478  ry = (y2 - y1)/2.0
479 
480  start = math.radians(float(start))
481  extent = math.radians(float(extent))
482 
483  # from SVG spec:
484  # http://www.w3.org/TR/SVG/implnote.html#ArcImplementationNotes
485  x1 = rx * math.cos(start) + cx
486  y1 = -ry * math.sin(start) + cy # XXX: ry is negated here
487 
488  x2 = rx * math.cos(start + extent) + cx
489  y2 = -ry * math.sin(start + extent) + cy # XXX: ry is negated here
490 
491  if abs(extent) > math.pi:
492  fa = 1
493  else:
494  fa = 0
495 
496  if extent > 0.0:
497  fs = 0
498  else:
499  fs = 1
500 
501  path = []
502  # common: arc
503  path.append('M%s,%s' % (x1, y1))
504  path.append('A%s,%s 0 %d %d %s,%s' % (rx, ry, fa, fs, x2, y2))
505 
506  if style == ARC:
507  pass
508 
509  elif style == CHORD:
510  path.append('z')
511 
512  else: # default: pieslice
513  path.append('L%s,%s' % (cx, cy))
514  path.append('z')
515 
516  return setattribs(document.createElement('path'), d = ''.join(path))

550 def arrow_head(document, x0, y0, x1, y1, arrowshape):
551  "make arrow head at (x1,y1), arrowshape is tuple (d1, d2, d3)"
552  import math
553 
554  dx = x1 - x0
555  dy = y1 - y0
556 
557  poly = document.createElement('polygon')
558 
559  d = math.sqrt(dx*dx + dy*dy)
560  if d == 0.0: # XXX: equal, no "close enough"
561  return poly
562 
563  d1, d2, d3 = list(map(float, arrowshape))
564  P0 = (x0, y0)
565  P1 = (x1, y1)
566 
567  xa, ya = lerp(P1, P0, d1/d)
568  xb, yb = lerp(P1, P0, d2/d)
569 
570  t = d3/d
571  xc, yc = dx*t, dy*t
572 
573  points = [
574  x1, y1,
575  xb - yc, yb + xc,
576  xa, ya,
577  xb + yc, yb - xc,
578  ]
579  poly.setAttribute('points', ' '.join(map(str, points)))
580  return poly

600 def parse_dash(string, width):
601  "parse dash pattern specified with string"
602 
603  # DashConvert from {tk-sources}/generic/tkCanvUtil.c
604  w = max(1, int(width + 0.5))
605 
606  n = len(string)
607  result = []
608  for i, c in enumerate(string):
609  if c == " " and len(result):
610  result[-1] += w + 1
611  elif c == "_":
612  result.append(8*w)
613  result.append(4*w)
614  elif c == "-":
615  result.append(6*w)
616  result.append(4*w)
617  elif c == ",":
618  result.append(4*w)
619  result.append(4*w)
620  elif c == ".":
621  result.append(2*w)
622  result.append(4*w)
