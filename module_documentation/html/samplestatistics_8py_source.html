43  def geometricMean(self):
44  '''
45  Calculates the geometric mean of the data
46 
47  @status: Tested method
48  @since: version 0.1
49  '''
50  mult = 1.0
51  one_over_n = 1.0 / len(self.data)
52  for item in self.data: mult = mult * math.pow(item, one_over_n)
53  return mult

55  def harmonicMean(self):
56  '''
57  Calculates the harmonic mean of the data

59  @status: Tested method
60  @since: version 0.1
61  '''
62  sum = 0.000001
63  for item in self.data:
64  if item != 0: sum = sum + 1.0/item

77  def moment(self, moment=1):
78  '''
79  Calculates the nth moment about the mean for the data
80  '''
81  if moment == 1:
82  return 0.0
83  else:
84  mn = self.arithmeticMean()
85  n = len(self.data)
86  s = 0
87  for x in self.data:
88  s = s + (x - mn) ** moment
89  return s / float(n)

188  def covariance(self):
189  '''
190  Calculates covariance using the formula: Cov(xy) = E(xy) - E(x)E(y)
191 
192  @status: Tested method
193  @since: version 0.3
194  '''
195  sname = self.listSamples()
196  X = self.sample[sname[0]]
197  Y = self.sample[sname[1]]
198  if X.data == Y.data: return 1.0
199  if X.rowcount == Y.rowcount: slen = X.rowcount
200  elif X.rowcount > Y.rowcount: slen = Y.rowcount
201  else: slen = X.rowcount
202  xy = SingleSample([X.data[i] * Y.data[i] for i in range(slen)],
203  'temporary')
204  mean_xy = xy.arithmeticMean()
205  mean_x = X.arithmeticMean()
206  mean_y = Y.arithmeticMean()
207  return mean_xy - (mean_x * mean_y)

209  def linear_regression(self):
210  '''
211  Calculates the first order linear regression model in the form of
212  "y = mx + c" from the 2 samples where the first sample (data1 and
213  name1 in initialization method) is taken as "X" and the second
214  sample (data2 and name2 in initialization method is taken as "Y".
215 
216  @return: Tuple of (gradient, intercept)
217 
218  @status: Tested method
219  @since: version 0.1
220  '''
221  sname = self.listSamples()
222  X = self.sample[sname[0]]
223  Y = self.sample[sname[1]]
224  if X.rowcount == Y.rowcount: slen = X.rowcount
225  elif X.rowcount > Y.rowcount: slen = Y.rowcount
226  else: slen = X.rowcount
227  mean_x = X.arithmeticMean()
228  mean_y = Y.arithmeticMean()
229  error_x = [X.data[i] - mean_x for i in range(slen)]
230  error_y = [Y.data[i] - mean_y for i in range(slen)]
231  gradient = sum([error_x[index] * error_y[index]
232  for index in range(len(error_x))]) / \
233  sum([error_x[index] * error_x[index]
234  for index in range(len(error_x))])
235  intercept = mean_y - (gradient * mean_x)
236  return (gradient, intercept)

238  def mlr(self, order=2):
239  sname = self.listSamples()
240  X = self.sample[sname[0]]
241  Y = self.sample[sname[1]]
242  if X.rowcount == Y.rowcount: slen = X.rowcount
243  elif X.rowcount > Y.rowcount: slen = Y.rowcount
244  else: slen = X.rowcount
245  X_data = X.data[:slen]
246  Y_data = Y.data[:slen]
247  data_array = [[1]*slen]
248  for x in range(1, order+1):
249  data_array.append([element**x for element in X_data])
250  beta_matrix = Matrix(data_array)
251  b1 = beta_matrix * beta_matrix.transpose()
252  b1 = b1.inverse()
253  b2 = Matrix(data_array) * Matrix([Y_data]).transpose()
254  result = b1 * b2
255  return [x[0] for x in result.m]

257  def pearson(self):
258  '''
259  Calculates the Pearson's product-moment coefficient by the formula
260 
261  (N * sum_xy) - (sum_x * sum_y)
262  --------------------------------------------------------------
263  ((N * sum_x2 - (sum_x)**2) * (N * sum_y2 - (sum_y)**2)) ** 0.5
264 
265  @status: Tested method
266  @since: version 0.1
267  '''
268  sname = self.listSamples()
269  X = self.sample[sname[0]]
270  Y = self.sample[sname[1]]
271  if X.rowcount == Y.rowcount: slen = X.rowcount
272  elif X.rowcount > Y.rowcount: slen = Y.rowcount
273  else: slen = X.rowcount
274  sum_x = sum([X.data[i] for i in range(slen)])
275  sum_x2 = sum([X.data[i] ** 2 for i in range(slen)])
276  sum_y = sum([Y.data[i] for i in range(slen)])
277  sum_y2 = sum([Y.data[i] ** 2 for i in range(slen)])
278  sum_xy = sum([X.data[i] * Y.data[i] for i in range(slen)])
279  numerator = (slen * sum_xy) - (sum_x * sum_y)
280  denominator_x = (slen * sum_x2) - (sum_x * sum_x)
281  denominator_y = (slen * sum_y2) - (sum_y * sum_y)
282  return float(numerator / ((denominator_x * denominator_y) ** 0.5))
