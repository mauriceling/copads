81 class MersenneTwister(Randomizer):
82  '''!
83  32-bit Mersenne twister algorithm (MT19937).
84 
85  Adapted from https://en.wikipedia.org/wiki/Mersenne_Twister
86 
87  Reference: Matsumoto, M., Nishimura, T. (1998). Mersenne twister:
88  a 623-dimensionally equidistributed uniform pseudo-random number
89  generator. ACM Transactions on Modeling and Computer Simulation 8
90  (1): 3-30. doi:10.1145/272991.272995
91  '''
92  def __init__(self, seed=None):
93  '''!
94  Constructor method.
95 
96  @param seed integer: seed to start the RNG. Default = None,
97  a random seed will be generated.
98  '''
99  if seed == None:
100  self.seed = int(randgen.random()*1000000)
101  else:
102  self.seed = int(seed)
103  self.index = 624
104  self.block = [0] * 624
105  self.block[0] = int(self.seed)
106  for i in range(1, 624):
107  t = self.block[i-1] ^ (self.block[i-1] >> 30)
108  self.block[i] = self._int32(1812433253 * t + i)
109 
110  def _int32(self, x):
111  '''!
112  Private method to get the 32 least significant bits.
113  '''
114  return int(0xFFFFFFFF & x)
115 
116  def random(self):
117  '''!
118  Method to generate a random integer.
119 
120  @return a random generated integer.
121  '''
122  if self.index >= 624: self._twist()
123  y = self.block[self.index]
124  # Right s+hift by 11 bits
125  y = y ^ (y >> 11)
126  # Shift y left by 7 and take the bitwise and of 2636928640
127  y = y ^ ((y << 7) & 2636928640)
128  # Shift y left by 15 and take the bitwise and of y and 4022730752
129  y = y ^ ((y << 15) & 4022730752)
130  # Right shift by 18 bits
131  y = y ^ (y >> 18)
132  self.index = self.index + 1
133  return self._int32(y)
134 
135  def _twist(self):
136  '''!
137  Private method to generate twist.
138  '''
139  for i in range(624):
140  # Get the most significant bit and add it to the less significant
141  # bits of the next number
142  t = self.block[(i+1) % 624] & 0x7fffffff
143  y = self._int32((self.block[i] & 0x80000000) + t)
144  self.block[i] = self.block[(i+397) % 624] ^ (y >> 1)
145  if y % 2 != 0:
146  self.block[i] = self.block[i] ^ 0x9908b0df
147  self.index = 0

150 class LCG(Randomizer):
151  '''!
152  A set of linear congruential generators (LCG) and LCG-based generators to generate a sequence of pseudorandom numbers. LCG has the general equation of
153 
154  x(n+1) = [multiplier * x(n) + increment] % modulus
155 
156  where
157 
158  1. increment (also known as offset) and modulus are co-primes
159  2. (multiplier - 1) is divisible by all prime factors of modulus
160  3. (multiplier - 1) is divisible by 4 if modulus is divisible by 4
161 
162  Depending on the parameters, different LCGs exist. If increment is
163  zero, the LCG is known as multiplicative congruential generator (
164  MCG). If increment is not zero, the LCG is known as mixed
165  congruential generator.
166  '''
167  def __init__(self, seed=None, generator='mmix'):
168  '''!
169  Constructor method.
170

172  a random seed will be generated.

174  - ansic: ANSI C (32-bit)

288  def random(self):
289  '''!
290  Method to generate a random integer using the following
291  equation where s is a newly generated integer
292 
293  x(n+1) = [multiplierX * x(n) + incrementX] % modulusX
294 
295  y(n+1) = [multiplierY * y(n) + incrementY] % modulusY
296 
297  s = [x(n+1) + y(n+1)] % max(modulusX, modulusY)
298 
299  @return a random generated integer.
300  '''
301  rA = self.LCG_A.random()
302  rB = self.LCG_B.random()
303  t = (rA + rB) % self.modulus
304  return t * self.modulus
