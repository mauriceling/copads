386 def betacf(a, b, x):
387  """
388  Continued fraction for incomplete beta function.
389  Adapted from salstat_stats.py of SalStat (www.sf.net/projects/salstat)
390  @see: NRP 6.3
391  @see: Ling, MHT. 2009. Compendium of Distributions, I: Beta, Binomial, Chi-
392  Square, F, Gamma, Geometric, Poisson, Student's t, and Uniform. The Python
393  Papers Source Codes 1:4
394  """
395  iter_max = 200
396  eps = 3.0e-7
397 
398  bm = az = am = 1.0
399  qab = a + b
400  qap = a + 1.0
401  qam = a - 1.0
402  bz = 1.0 - qab * x / qap
403  for i in range(iter_max + 1):
404  em = float(i + 1)
405  tem = em + em
406  d = em * (b - em) * x / ((qam + tem) * (a + tem))
407  ap = az + d * am
408  bp = bz + d * bm
409  d = -(a + em) * (qab + em) * x / ((qap + tem) * (a + tem))
410  app = ap + d * az
411  bpp = bp + d * bz
412  aold = az
413  am = ap / bpp
414  bm = bp / bpp
415  az = app / bpp
416  bz = 1.0
417  if (abs(az - aold) < (eps * abs(az))):
418  return az

420 def betai(a, b, x):
421  """
422  Incomplete beta function
423 
424  I-sub-x(a,b) = 1/B(a,b)*(Integral(0,x) of t^(a-1)(1-t)^(b-1) dt)
425 
426  where a,b>0 and B(a,b) = G(a)*G(b)/(G(a+b)) where G(a) is the gamma
427  function of a.
428 
429  Adapted from salstat_stats.py of SalStat (www.sf.net/projects/salstat)
430  Depend: betacf, gammln
431  @see: NRP 6.3
432 
433  @see: Ling, MHT. 2009. Compendium of Distributions, I: Beta, Binomial, Chi-
434  Square, F, Gamma, Geometric, Poisson, Student's t, and Uniform. The Python
435  Papers Source Codes 1:4
436 
437  @status: Tested function
438  @since: version 0.1
439  """
440  if (x < 0.0 or x > 1.0):
441  raise ValueError('Bad value for x: %s' % x)
442  if (x == 0.0 or x == 1.0):
443  bt = 0.0
444  else:
445  bt = math.exp(gammln(a+b) - gammln(a) - gammln(b) + a *
446  math.log(x) + b * math.log(1.0-x))
447  if (x < (a + 1.0) / (a + b + 2.0)):
448  return bt * betacf(a, b, x) / float(a)
449  else:
450  return 1.0 - bt * betacf(b, a, 1.0 - x) / float(b)

673 def gcf(a, x, itmax=200, eps=3.e-7):
674  """

676  @see: http://mail.python.org/pipermail/python-list/2000-June/671838.html
677

682  @status: Tested function
683  @since: version 0.1
684  """

687  a0 = 1.0
688  a1 = x
689  b0 = 0.0
690  b1 = 1.0

692  n = 1
693  while n <= itmax:
694  an = n
695  ana = an - a
696  a0 = (a1 + a0 * ana) * fac
697  b0 = (b1 + b0 * ana) * fac
698  anf = an * fac

700  b1 = x * b0 + anf * b1

702  fac = 1.0 / a1

705  return (g * math.exp(-x + a * math.log(x) - gln), gln)
706  gold = g
707  n = n + 1

709  % abs((g - gold) / g))

809 
810  @param data: a 1-dimensional list of numerical values

816  - skew = skew
817  - kurt = kurtosis
818  """

827  s = d - ave
